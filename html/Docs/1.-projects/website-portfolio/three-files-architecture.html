import * as THREE from 'three';

/** @returns {THREE.WebGLRenderer} */
export function createRenderer() {
  const r = new THREE.WebGLRenderer({ antialias: true });
  r.setPixelRatio(Math.min(window.devicePixelRatio, 2));
  return r;
}
